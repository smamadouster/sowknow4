```python
# guardian_hc/plugin.py
"""Guardian v2 plugin interface and shared types."""

from __future__ import annotations

import enum
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any


class Severity(enum.IntEnum):
    """Alert severity levels, ordered for comparison."""
    INFO = 0
    WARNING = 1
    HIGH = 2
    CRITICAL = 3


@dataclass
class CheckContext:
    """Context passed to plugin.check() each patrol cycle."""
    patrol_level: str  # "critical", "standard", "deep"
    config: dict[str, Any]
    services: list[Any]  # ServiceConfig list
    metrics_db: Any = None  # MetricsDB instance (for sentinel/trends)


@dataclass
class AnalysisContext:
    """Context passed to plugin.analyze() on deep patrol."""
    config: dict[str, Any]
    metrics_db: Any = None
    patterns_db: Any = None
    recent_incidents: list[dict] = field(default_factory=list)


@dataclass
class CheckResult:
    """Result from a single health check."""
    plugin: str
    module: str
    check_name: str
    status: str  # "pass", "fail", "warning", "degraded"
    severity: Severity
    summary: str
    details: dict[str, Any] = field(default_factory=dict)
    needs_healing: bool = False
    heal_hint: str | None = None
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


@dataclass
class HealResult:
    """Result from a healing action."""
    plugin: str
    target: str
    action: str
    success: bool
    details: str = ""
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


@dataclass
class Insight:
    """Result from trend analysis or pattern matching."""
    plugin: str
    insight_type: str  # "prediction", "anomaly", "capacity", "pattern_match"
    severity: Severity
    summary: str
    metric: str = ""
    current_value: float | None = None
    predicted_value: float | None = None
    predicted_time_hours: float | None = None
    recommended_action: str | None = None
    pattern_name: str | None = None
    confidence: float | None = None
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


class GuardianPlugin:
    """Base class for all Guardian plugins.

    Subclasses must implement check(), and optionally heal() and analyze().
    """
    name: str = "unnamed"
    enabled: bool = True

    async def check(self, context: CheckContext) -> list[CheckResult]:
        """Run health checks. Called every patrol cycle."""
        return []

    async def heal(self, result: CheckResult) -> HealResult | None:
        """Attempt to heal a failed check. Called when check returns needs_healing=True."""
        return None

    async def analyze(self, context: AnalysisContext) -> list[Insight]:
        """Run analysis (trends, patterns). Called on deep patrol only."""
        return []
```

```python
# guardian_hc/db.py
"""Guardian v2 database layer — PostgreSQL with SQLite fallback."""

from __future__ import annotations

import json
import math
import time
from datetime import datetime, timezone, timedelta
from typing import Any

import aiosqlite
import structlog

logger = structlog.get_logger()

# Try asyncpg import — optional for testing with SQLite-only
try:
    import asyncpg
except ImportError:
    asyncpg = None

PG_SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS guardian_metrics (
    id BIGSERIAL PRIMARY KEY,
    ts TIMESTAMPTZ NOT NULL DEFAULT now(),
    metric VARCHAR(64) NOT NULL,
    service VARCHAR(32),
    value DOUBLE PRECISION NOT NULL,
    tags JSONB DEFAULT '{}'
);
CREATE INDEX IF NOT EXISTS idx_gm_metric_ts ON guardian_metrics (metric, ts DESC);

CREATE TABLE IF NOT EXISTS guardian_metrics_hourly (
    id BIGSERIAL PRIMARY KEY,
    ts TIMESTAMPTZ NOT NULL,
    metric VARCHAR(64) NOT NULL,
    service VARCHAR(32),
    avg_value DOUBLE PRECISION NOT NULL,
    min_value DOUBLE PRECISION NOT NULL,
    max_value DOUBLE PRECISION NOT NULL,
    sample_count INT NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_gmh_metric_ts ON guardian_metrics_hourly (metric, ts DESC);

CREATE TABLE IF NOT EXISTS guardian_patterns (
    id SERIAL PRIMARY KEY,
    pattern_name VARCHAR(128) NOT NULL,
    trigger_conditions JSONB NOT NULL,
    predicted_outcome VARCHAR(256),
    recommended_action VARCHAR(256),
    confidence DOUBLE PRECISION DEFAULT 0.5,
    times_matched INT DEFAULT 0,
    times_correct INT DEFAULT 0,
    last_matched TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT now(),
    active BOOLEAN DEFAULT true
);
"""

SQLITE_SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS guardian_metrics (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    ts TEXT NOT NULL,
    metric TEXT NOT NULL,
    service TEXT,
    value REAL NOT NULL,
    tags TEXT DEFAULT '{}'
);
CREATE INDEX IF NOT EXISTS idx_gm_metric_ts ON guardian_metrics (metric, ts);

CREATE TABLE IF NOT EXISTS guardian_patterns (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    pattern_name TEXT NOT NULL,
    trigger_conditions TEXT NOT NULL,
    predicted_outcome TEXT,
    recommended_action TEXT,
    confidence REAL DEFAULT 0.5,
    times_matched INTEGER DEFAULT 0,
    times_correct INTEGER DEFAULT 0,
    last_matched TEXT,
    created_at TEXT DEFAULT (datetime('now')),
    active INTEGER DEFAULT 1
);
"""


class MetricsDB:
    """Unified metrics/patterns DB with PostgreSQL primary and SQLite fallback."""

    def __init__(self, pg_dsn: str | None = None, fallback_path: str = "/tmp/guardian-metrics-buffer.db"):
        self._pg_dsn = pg_dsn
        self._fallback_path = fallback_path
        self._pg_pool: Any = None
        self._sqlite: aiosqlite.Connection | None = None
        self._using_pg = False

    async def connect(self):
        """Connect to PostgreSQL, fall back to SQLite on failure."""
        if self._pg_dsn and asyncpg:
            try:
                self._pg_pool = await asyncpg.create_pool(
                    self._pg_dsn, min_size=1, max_size=3, command_timeout=10
                )
                async with self._pg_pool.acquire() as conn:
                    await conn.execute(PG_SCHEMA_SQL)
                self._using_pg = True
                logger.info("db.connected", backend="postgresql")
                return
            except Exception as e:
                logger.warning("db.pg_failed", error=str(e)[:200])

        # Fallback to SQLite
        self._sqlite = await aiosqlite.connect(self._fallback_path)
        self._sqlite.row_factory = aiosqlite.Row
        await self._sqlite.executescript(SQLITE_SCHEMA_SQL)
        self._using_pg = False
        logger.info("db.connected", backend="sqlite", path=self._fallback_path)

    async def close(self):
        if self._pg_pool:
            await self._pg_pool.close()
        if self._sqlite:
            await self._sqlite.close()

    @property
    def is_pg(self) -> bool:
        return self._using_pg

    # --- Metrics ---

    async def write_metric(self, metric: str, value: float, service: str = "", tags: dict | None = None):
        now = datetime.now(timezone.utc).isoformat()
        tags_json = json.dumps(tags or {})
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                await conn.execute(
                    "INSERT INTO guardian_metrics (ts, metric, service, value, tags) VALUES ($1, $2, $3, $4, $5::jsonb)",
                    datetime.now(timezone.utc), metric, service, value, tags_json,
                )
        else:
            await self._sqlite.execute(
                "INSERT INTO guardian_metrics (ts, metric, service, value, tags) VALUES (?, ?, ?, ?, ?)",
                (now, metric, service, value, tags_json),
            )
            await self._sqlite.commit()

    async def write_batch(self, metrics: list[tuple[str, float, str, dict]]):
        """Write multiple metrics: [(metric, value, service, tags), ...]"""
        now = datetime.now(timezone.utc)
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                await conn.executemany(
                    "INSERT INTO guardian_metrics (ts, metric, service, value, tags) VALUES ($1, $2, $3, $4, $5::jsonb)",
                    [(now, m, s, v, json.dumps(t)) for m, v, s, t in metrics],
                )
        else:
            now_iso = now.isoformat()
            await self._sqlite.executemany(
                "INSERT INTO guardian_metrics (ts, metric, service, value, tags) VALUES (?, ?, ?, ?, ?)",
                [(now_iso, m, s, v, json.dumps(t)) for m, v, s, t in metrics],
            )
            await self._sqlite.commit()

    async def query_metrics(self, metric: str, hours: int = 24, service: str | None = None) -> list[dict]:
        cutoff = datetime.now(timezone.utc) - timedelta(hours=hours)
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                if service:
                    rows = await conn.fetch(
                        "SELECT ts, metric, service, value, tags FROM guardian_metrics WHERE metric=$1 AND service=$2 AND ts>=$3 ORDER BY ts DESC",
                        metric, service, cutoff,
                    )
                else:
                    rows = await conn.fetch(
                        "SELECT ts, metric, service, value, tags FROM guardian_metrics WHERE metric=$1 AND ts>=$2 ORDER BY ts DESC",
                        metric, cutoff,
                    )
                return [dict(r) for r in rows]
        else:
            cutoff_iso = cutoff.isoformat()
            if service:
                cursor = await self._sqlite.execute(
                    "SELECT ts, metric, service, value, tags FROM guardian_metrics WHERE metric=? AND service=? AND ts>=? ORDER BY ts DESC",
                    (metric, service, cutoff_iso),
                )
            else:
                cursor = await self._sqlite.execute(
                    "SELECT ts, metric, service, value, tags FROM guardian_metrics WHERE metric=? AND ts>=? ORDER BY ts DESC",
                    (metric, cutoff_iso),
                )
            rows = await cursor.fetchall()
            return [{"ts": r[0], "metric": r[1], "service": r[2], "value": r[3], "tags": r[4]} for r in rows]

    async def get_latest(self, metric: str, service: str | None = None) -> float | None:
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                if service:
                    row = await conn.fetchrow(
                        "SELECT value FROM guardian_metrics WHERE metric=$1 AND service=$2 ORDER BY ts DESC LIMIT 1",
                        metric, service,
                    )
                else:
                    row = await conn.fetchrow(
                        "SELECT value FROM guardian_metrics WHERE metric=$1 ORDER BY ts DESC LIMIT 1",
                        metric,
                    )
                return row["value"] if row else None
        else:
            if service:
                cursor = await self._sqlite.execute(
                    "SELECT value FROM guardian_metrics WHERE metric=? AND service=? ORDER BY ts DESC LIMIT 1",
                    (metric, service),
                )
            else:
                cursor = await self._sqlite.execute(
                    "SELECT value FROM guardian_metrics WHERE metric=? ORDER BY ts DESC LIMIT 1",
                    (metric,),
                )
            row = await cursor.fetchone()
            return row[0] if row else None

    async def get_slope(self, metric: str, hours: int = 6, service: str | None = None) -> float | None:
        """Linear regression slope over the time window. Returns value-per-hour."""
        rows = await self.query_metrics(metric, hours=hours, service=service)
        if len(rows) < 2:
            return None

        # Convert timestamps to hours-from-now for regression
        now = datetime.now(timezone.utc)
        points = []
        for r in rows:
            ts = r["ts"]
            if isinstance(ts, str):
                ts = datetime.fromisoformat(ts)
            if ts.tzinfo is None:
                ts = ts.replace(tzinfo=timezone.utc)
            hours_ago = (now - ts).total_seconds() / 3600.0
            points.append((hours_ago, r["value"]))

        # Simple linear regression: y = a + b*x
        n = len(points)
        sum_x = sum(p[0] for p in points)
        sum_y = sum(p[1] for p in points)
        sum_xy = sum(p[0] * p[1] for p in points)
        sum_x2 = sum(p[0] ** 2 for p in points)

        denom = n * sum_x2 - sum_x ** 2
        if abs(denom) < 1e-10:
            return 0.0

        slope = (n * sum_xy - sum_x * sum_y) / denom
        # Negate because hours_ago is inverted (larger = older)
        return -slope

    async def get_mean_stddev(self, metric: str, hours: int = 24, service: str | None = None) -> tuple[float, float]:
        """Return (mean, stddev) for a metric over the time window."""
        rows = await self.query_metrics(metric, hours=hours, service=service)
        if not rows:
            return 0.0, 0.0
        values = [r["value"] for r in rows]
        n = len(values)
        mean = sum(values) / n
        if n < 2:
            return mean, 0.0
        variance = sum((v - mean) ** 2 for v in values) / (n - 1)
        return mean, math.sqrt(variance)

    async def purge_raw_metrics(self, hours: int = 48) -> int:
        """Delete raw metrics older than `hours`. Returns count deleted."""
        cutoff = datetime.now(timezone.utc) - timedelta(hours=hours)
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                result = await conn.execute(
                    "DELETE FROM guardian_metrics WHERE ts < $1", cutoff,
                )
                return int(result.split()[-1])
        else:
            cutoff_iso = cutoff.isoformat()
            cursor = await self._sqlite.execute(
                "DELETE FROM guardian_metrics WHERE ts < ?", (cutoff_iso,),
            )
            await self._sqlite.commit()
            return cursor.rowcount

    # --- Patterns ---

    async def create_pattern(
        self,
        pattern_name: str,
        trigger_conditions: dict,
        predicted_outcome: str,
        recommended_action: str,
        confidence: float = 0.5,
    ) -> int:
        cond_json = json.dumps(trigger_conditions)
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                row = await conn.fetchrow(
                    "INSERT INTO guardian_patterns (pattern_name, trigger_conditions, predicted_outcome, recommended_action, confidence) "
                    "VALUES ($1, $2::jsonb, $3, $4, $5) RETURNING id",
                    pattern_name, cond_json, predicted_outcome, recommended_action, confidence,
                )
                return row["id"]
        else:
            cursor = await self._sqlite.execute(
                "INSERT INTO guardian_patterns (pattern_name, trigger_conditions, predicted_outcome, recommended_action, confidence) "
                "VALUES (?, ?, ?, ?, ?)",
                (pattern_name, cond_json, predicted_outcome, recommended_action, confidence),
            )
            await self._sqlite.commit()
            return cursor.lastrowid

    async def get_active_patterns(self) -> list[dict]:
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                rows = await conn.fetch(
                    "SELECT id, pattern_name, trigger_conditions, predicted_outcome, recommended_action, "
                    "confidence, times_matched, times_correct, last_matched, active "
                    "FROM guardian_patterns WHERE active=true ORDER BY confidence DESC"
                )
                return [dict(r) for r in rows]
        else:
            cursor = await self._sqlite.execute(
                "SELECT id, pattern_name, trigger_conditions, predicted_outcome, recommended_action, "
                "confidence, times_matched, times_correct, last_matched, active "
                "FROM guardian_patterns WHERE active=1 ORDER BY confidence DESC"
            )
            rows = await cursor.fetchall()
            return [
                {
                    "id": r[0], "pattern_name": r[1],
                    "trigger_conditions": json.loads(r[2]) if isinstance(r[2], str) else r[2],
                    "predicted_outcome": r[3], "recommended_action": r[4],
                    "confidence": r[5], "times_matched": r[6], "times_correct": r[7],
                    "last_matched": r[8], "active": bool(r[9]),
                }
                for r in rows
            ]

    async def update_pattern_confidence(self, pattern_id: int, matched: bool, correct: bool):
        """Update pattern confidence after a match."""
        now = datetime.now(timezone.utc)
        if self._using_pg:
            async with self._pg_pool.acquire() as conn:
                row = await conn.fetchrow(
                    "SELECT confidence, times_matched, times_correct FROM guardian_patterns WHERE id=$1",
                    pattern_id,
                )
                if not row:
                    return
                conf = row["confidence"]
                tm = row["times_matched"]
                tc = row["times_correct"]
        else:
            cursor = await self._sqlite.execute(
                "SELECT confidence, times_matched, times_correct FROM guardian_patterns WHERE id=?",
                (pattern_id,),
            )
            row = await cursor.fetchone()
            if not row:
                return
            conf, tm, tc = row[0], row[1], row[2]

        if matched:
            tm += 1
        if correct:
            tc += 1
            conf = min(conf + 0.1, 0.95)
        else:
```

```python
# guardian_hc/agents.py
"""Guardian v2 agent persona definitions and health tracking."""

from __future__ import annotations

import enum
from dataclasses import dataclass, field
from datetime import datetime, timezone


class AgentStatus(str, enum.Enum):
    HEALTHY = "healthy"
    DEGRADED = "degraded"
    UNHEALTHY = "unhealthy"


@dataclass
class Agent:
    """A named Guardian agent with health tracking."""
    agent_id: str
    name: str
    role: str
    plugins: list[str] = field(default_factory=list)
    status: AgentStatus = AgentStatus.HEALTHY
    checks_total: int = 0
    checks_failed: int = 0
    _recent_failures: int = 0

    def record_check(self, success: bool):
        self.checks_total += 1
        if not success:
            self.checks_failed += 1
            self._recent_failures += 1
        else:
            self._recent_failures = 0

        # 3 consecutive failures → degraded
        if self._recent_failures >= 3:
            self.status = AgentStatus.DEGRADED
        else:
            self.status = AgentStatus.HEALTHY

    def to_dict(self) -> dict:
        return {
            "id": self.agent_id,
            "name": self.name,
            "role": self.role,
            "health": self.status.value,
            "checks_total": self.checks_total,
            "checks_failed": self.checks_failed,
        }


class AgentRegistry:
    """Registry of all Guardian agents."""

    def __init__(self, config: list[dict]):
        self.agents: list[Agent] = []
        for c in config:
            self.agents.append(Agent(
                agent_id=c["id"],
                name=c["name"],
                role=c.get("role", ""),
                plugins=c.get("plugins", []),
            ))

    def get(self, agent_id: str) -> Agent | None:
        for a in self.agents:
            if a.agent_id == agent_id:
                return a
        return None

    def get_by_name(self, name: str) -> Agent | None:
        for a in self.agents:
            if a.name == name:
                return a
        return None

    def summary(self) -> list[dict]:
        return [a.to_dict() for a in self.agents]
```

```python
"""Guardian v2 configuration model — backward-compatible with v1."""

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any


@dataclass
class ServiceConfig:
    name: str
    container: str
    health_check: dict = field(default_factory=dict)
    auto_heal: dict = field(default_factory=dict)


@dataclass
class ModuleConfig:
    name: str
    services: list[str] = field(default_factory=list)
    probes: list[str] = field(default_factory=list)


@dataclass
class GuardianV2Config:
    version: str = "2.0"
    app_name: str = "Application"
    compose_file: str = "./docker-compose.yml"
    services: list[ServiceConfig] = field(default_factory=list)
    plugins: dict[str, dict] = field(default_factory=dict)
    modules: list[ModuleConfig] = field(default_factory=list)
    database: dict[str, Any] = field(default_factory=dict)
    agents: list[dict] = field(default_factory=list)
    alerts: dict = field(default_factory=dict)
    patrols: dict = field(default_factory=dict)
    disk: dict = field(default_factory=dict)
    ssl: dict = field(default_factory=dict)
    ollama: dict = field(default_factory=dict)
    vps_load: dict = field(default_factory=dict)
    network: dict = field(default_factory=dict)
    celery: dict = field(default_factory=dict)
    daily_report: dict = field(default_factory=dict)
    dashboard_port: int = 9090


def load_config(raw: dict) -> GuardianV2Config:
    """Parse raw YAML dict into GuardianV2Config. Supports v1 and v2 formats."""
    version = raw.get("version", "1.0")

    services = []
    for svc in raw.get("services", []):
        services.append(ServiceConfig(
            name=svc.get("name", ""),
            container=svc.get("container", ""),
            health_check=svc.get("health_check", {}),
            auto_heal=svc.get("auto_heal", {}),
        ))

    modules = []
    for mod in raw.get("modules", []):
        modules.append(ModuleConfig(
            name=mod.get("name", ""),
            services=mod.get("services", []),
            probes=mod.get("probes", []),
        ))

    return GuardianV2Config(
        version=version,
        app_name=raw.get("app", {}).get("name", "Application"),
        compose_file=raw.get("app", {}).get("compose_file", "./docker-compose.yml"),
        services=services,
        plugins=raw.get("plugins", {}),
        modules=modules,
        database=raw.get("database", {}),
        agents=raw.get("agents", []),
        alerts=raw.get("alerts", {}),
        patrols=raw.get("patrols", {}),
        disk=raw.get("disk", {}),
        ssl=raw.get("ssl", {}),
        ollama=raw.get("ollama", {}),
        vps_load=raw.get("vps_load", {}),
        network=raw.get("network", {}),
        celery=raw.get("celery", {}),
```

```python
        """Register a plugin by name."""
        self._plugins[plugin.name] = plugin
        logger.info("plugin.registered", name=plugin.name)

    async def run_plugin_checks(self, level: str) -> list[CheckResult]:
        """Run check() on all enabled plugins for the given patrol level."""
        context = CheckContext(
            patrol_level=level,
            config={},
            services=self.config.services if hasattr(self.config, 'services') else [],
            metrics_db=self._metrics_db,
        )
        all_results: list[CheckResult] = []
        for name, plugin in self._plugins.items():
            if not plugin.enabled:
                continue
            try:
                results = await plugin.check(context)
                all_results.extend(results)
            except Exception as e:
                logger.error("plugin.check.error", plugin=name, error=str(e)[:200])
        return all_results

    async def run_plugin_heals(self, check_results: list[CheckResult]) -> list[HealResult]:
        """Run heal() for all failed checks that need healing."""
        heal_results: list[HealResult] = []
        for result in check_results:
            if not result.needs_healing:
                continue
            plugin = self._plugins.get(result.plugin)
            if not plugin:
                continue
            try:
                heal = await plugin.heal(result)
                if heal:
                    heal_results.append(heal)
                    self.log_action({"type": "heal", "plugin": result.plugin, "target": heal.target, "action": heal.action, "success": heal.success})
            except Exception as e:
                logger.error("plugin.heal.error", plugin=result.plugin, error=str(e)[:200])
        return heal_results

    async def run_plugin_analysis(self) -> list[Insight]:
        """Run analyze() on all enabled plugins. Called on deep patrol."""
        context = AnalysisContext(
            config={},
```

```python
# guardian_hc/plugins/infrastructure.py
"""Infrastructure plugin — wraps existing v1 checkers and healers."""

from __future__ import annotations

import structlog

from guardian_hc.plugin import (
    GuardianPlugin, CheckResult, HealResult, CheckContext, Severity,
)
from guardian_hc.checks.containers import ContainerChecker
from guardian_hc.checks.http_health import HttpHealthChecker
from guardian_hc.checks.tcp_health import TcpHealthChecker
from guardian_hc.checks.disk import DiskChecker
from guardian_hc.checks.memory import MemoryChecker
from guardian_hc.checks.ssl_check import SslChecker
from guardian_hc.checks.config_drift import ConfigDriftChecker
from guardian_hc.checks.network_health import NetworkHealthChecker
from guardian_hc.checks.celery_health import CeleryHealthChecker
from guardian_hc.checks.vps_load import VpsLoadChecker
from guardian_hc.checks.ollama_health import OllamaChecker
from guardian_hc.healers.container_healer import ContainerHealer
from guardian_hc.healers.disk_healer import DiskHealer
from guardian_hc.healers.ssl_healer import SslHealer
from guardian_hc.healers.memory_healer import MemoryHealer
from guardian_hc.healers.network_healer import NetworkHealer
from guardian_hc.config import ServiceConfig

logger = structlog.get_logger()

# Checks to run per patrol level
PATROL_CHECKS = {
    "critical": ["containers", "api_health"],
    "standard": ["containers", "api_health", "disk", "memory", "network", "celery", "vps_load"],
    "deep": ["containers", "api_health", "disk", "memory", "network", "celery", "vps_load", "ssl", "config_drift"],
}


class InfrastructurePlugin(GuardianPlugin):
    name = "infrastructure"
    enabled = True

    def __init__(self, config: dict):
        self._services: list[ServiceConfig] = config.get("services", [])
        self._container_checker = ContainerChecker()
        self._container_healer = ContainerHealer()
        self._http_checker = HttpHealthChecker
        self._tcp_checker = TcpHealthChecker
        self._disk_checker = DiskChecker(config.get("disk", {}))
        self._disk_healer = DiskHealer(config.get("disk", {}))
        self._memory_checker = MemoryChecker()
        self._memory_healer = MemoryHealer()
        self._ssl_checker = SslChecker(config.get("ssl", {}))
        self._ssl_healer = SslHealer(config.get("ssl", {}))
        self._drift_checker = ConfigDriftChecker(config)
        self._network_checker = NetworkHealthChecker(config.get("network", {}))
        self._network_healer = NetworkHealer({"compose_file": config.get("compose_file", "")})
        self._celery_checker = CeleryHealthChecker(config.get("celery", {}))
        self._vps_load_checker = VpsLoadChecker(config.get("vps_load", {}))
        self._ollama_checker = OllamaChecker(config.get("ollama", {}))
        self._compose_file = config.get("compose_file", "")

    async def check(self, context: CheckContext) -> list[CheckResult]:
        level = context.patrol_level
        checks_to_run = PATROL_CHECKS.get(level, PATROL_CHECKS["standard"])
        results: list[CheckResult] = []

        # Container + API health checks
        if "containers" in checks_to_run or "api_health" in checks_to_run:
            for svc in self._services:
                container_status = await self._container_checker.check(svc.container)
                running = container_status.get("status") == "running"

                if not running:
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name=f"container.{svc.name}",
                        status="fail", severity=Severity.CRITICAL,
                        summary=f"{svc.name} container not running",
                        details=container_status,
                        needs_healing=svc.auto_heal.get("restart", False),
                        heal_hint=f"restart:{svc.container}",
                    ))
                    continue

                # HTTP/TCP health check
                hc = svc.health_check
                if hc.get("type") == "http" and "api_health" in checks_to_run:
                    http_result = await self._http_checker.check(
                        hc.get("url", ""), timeout=hc.get("timeout", 10)
                    )
                    if not http_result.get("healthy"):
                        results.append(CheckResult(
                            plugin=self.name, module="Infrastructure",
                            check_name=f"http.{svc.name}",
                            status="fail", severity=Severity.HIGH,
                            summary=f"{svc.name} HTTP health check failed",
                            details=http_result,
                            needs_healing=svc.auto_heal.get("restart", False),
                            heal_hint=f"restart:{svc.container}",
                        ))
                    else:
                        results.append(CheckResult(
                            plugin=self.name, module="Infrastructure",
                            check_name=f"http.{svc.name}",
                            status="pass", severity=Severity.INFO,
                            summary=f"{svc.name} healthy",
                            details=http_result,
                        ))

                elif hc.get("type") == "tcp":
                    tcp_result = await self._tcp_checker.check(
                        hc.get("host", ""), hc.get("port", 0), timeout=hc.get("timeout", 5)
                    )
                    if not tcp_result.get("healthy"):
                        results.append(CheckResult(
                            plugin=self.name, module="Infrastructure",
                            check_name=f"tcp.{svc.name}",
                            status="fail", severity=Severity.HIGH,
                            summary=f"{svc.name} TCP health check failed",
                            details=tcp_result,
                            needs_healing=svc.auto_heal.get("restart", False),
                            heal_hint=f"restart:{svc.container}",
                        ))
                    else:
                        results.append(CheckResult(
                            plugin=self.name, module="Infrastructure",
                            check_name=f"tcp.{svc.name}",
                            status="pass", severity=Severity.INFO,
                            summary=f"{svc.name} healthy",
                            details=tcp_result,
                        ))
                else:
                    # Container-only check (type: container) — running is enough
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name=f"container.{svc.name}",
                        status="pass", severity=Severity.INFO,
                        summary=f"{svc.name} running",
                        details=container_status,
                    ))

        # Disk check
        if "disk" in checks_to_run:
            try:
                disk = await self._disk_checker.check()
                severity = Severity.INFO
                status = "pass"
                if disk.get("severity") == "critical":
                    severity = Severity.CRITICAL
                    status = "fail"
                elif disk.get("severity") == "warning":
                    severity = Severity.WARNING
                    status = "warning"
                results.append(CheckResult(
                    plugin=self.name, module="Infrastructure",
                    check_name="disk", status=status, severity=severity,
                    summary=f"Disk usage: {disk.get('usage_pct', '?')}%",
                    details=disk,
                    needs_healing=disk.get("needs_healing", False),
                    heal_hint="disk_cleanup",
                ))
            except Exception as e:
                logger.error("infra.disk.error", error=str(e)[:200])

        # Memory check
        if "memory" in checks_to_run:
            try:
                mem_results = await self._memory_checker.check(self._services)
                for m in mem_results:
                    needs_heal = m.get("needs_healing", False)
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name=f"memory.{m.get('container', 'unknown')}",
                        status="fail" if needs_heal else "pass",
                        severity=Severity.HIGH if needs_heal else Severity.INFO,
                        summary=f"{m.get('container')}: {m.get('mem_pct', '?')}% memory",
                        details=m,
                        needs_healing=needs_heal,
                        heal_hint=f"restart:{m.get('container')}" if needs_heal else None,
                    ))
            except Exception as e:
                logger.error("infra.memory.error", error=str(e)[:200])

        # Network check
        if "network" in checks_to_run:
            try:
                net = await self._network_checker.check()
                needs_heal = net.get("needs_healing", False)
                results.append(CheckResult(
                    plugin=self.name, module="Infrastructure",
                    check_name="network", status="fail" if needs_heal else "pass",
                    severity=Severity.CRITICAL if needs_heal else Severity.INFO,
                    summary="Network degraded" if needs_heal else "Network healthy",
                    details=net,
                    needs_healing=needs_heal,
                    heal_hint="network_heal" if needs_heal else None,
                ))
            except Exception as e:
                logger.error("infra.network.error", error=str(e)[:200])

        # Celery check
        if "celery" in checks_to_run:
            try:
                celery_results = await self._celery_checker.check()
                for c in celery_results:
                    needs_heal = c.get("needs_healing", False)
                    check_type = c.get("check", "celery")
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name=f"celery.{check_type}",
                        status="fail" if needs_heal else "pass",
                        severity=Severity.HIGH if needs_heal else Severity.INFO,
                        summary=c.get("detail", c.get("status", "")),
                        details=c,
                        needs_healing=needs_heal,
                        heal_hint=f"restart:{c.get('container')}" if needs_heal and c.get("container") else None,
                    ))
            except Exception as e:
                logger.error("infra.celery.error", error=str(e)[:200])

        # VPS load check
        if "vps_load" in checks_to_run:
            try:
                load_results = await self._vps_load_checker.check()
                for lr in load_results:
                    needs_heal = lr.get("needs_healing", False)
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name=f"vps.{lr.get('type', 'load')}",
                        status="warning" if needs_heal else "pass",
                        severity=Severity.WARNING if needs_heal else Severity.INFO,
                        summary=f"VPS {lr.get('type')}: {lr.get('load1', lr.get('steal_pct', '?'))}",
                        details=lr,
                    ))
            except Exception as e:
                logger.error("infra.vps.error", error=str(e)[:200])

        # SSL check
        if "ssl" in checks_to_run:
            try:
                ssl_results = await self._ssl_checker.check()
                for s in ssl_results:
                    needs_heal = s.get("needs_healing", False)
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name=f"ssl.{s.get('domain', 'unknown')}",
                        status="fail" if needs_heal else "pass",
                        severity=Severity.HIGH if needs_heal else Severity.INFO,
                        summary=f"SSL {s.get('domain')}: {s.get('days_left', '?')} days left",
                        details=s,
                        needs_healing=needs_heal,
                        heal_hint=f"ssl_renew:{s.get('domain')}" if needs_heal else None,
                    ))
            except Exception as e:
                logger.error("infra.ssl.error", error=str(e)[:200])

        # Config drift
        if "config_drift" in checks_to_run:
            try:
                drift = await self._drift_checker.check()
                drift_count = drift.get("count", 0)
                if drift_count > 0:
                    results.append(CheckResult(
                        plugin=self.name, module="Infrastructure",
                        check_name="config_drift",
                        status="warning", severity=Severity.WARNING,
                        summary=f"{drift_count} config drift(s) detected",
                        details=drift,
                    ))
            except Exception as e:
                logger.error("infra.drift.error", error=str(e)[:200])

        return results

    async def heal(self, result: CheckResult) -> HealResult | None:
        hint = result.heal_hint or ""

        if hint.startswith("restart:"):
            container = hint.split(":", 1)[1]
            # Find service config for rebuild flag
            rebuild = False
            for svc in self._services:
                if svc.container == container:
                    rebuild = svc.auto_heal.get("rebuild_on_failure", False)
                    break
            heal = await self._container_healer.heal(
                container, rebuild=rebuild, compose_file=self._compose_file
            )
            return HealResult(
                plugin=self.name, target=container,
                action=heal.get("action", "restart"),
                success=heal.get("healed", False),
                details=str(heal),
            )

        if hint == "disk_cleanup":
            heal = await self._disk_healer.heal()
            return HealResult(
                plugin=self.name, target="disk",
                action="cleanup",
                success=heal.get("healed", False),
                details=str(heal.get("actions", [])),
            )

        if hint == "network_heal":
            stale = result.details.get("stale_bridges", [])
            heal = await self._network_healer.heal(stale_bridges=stale)
            return HealResult(
                plugin=self.name, target="network",
                action="network_repair",
                success=heal.get("healed", False),
                details=str(heal.get("actions", [])),
            )

        if hint and hint.startswith("ssl_renew:"):
            domain = hint.split(":", 1)[1]
            heal = await self._ssl_healer.heal(domain)
            return HealResult(
                plugin=self.name, target=f"ssl:{domain}",
                action="ssl_renew",
                success=heal.get("healed", False),
                details=heal.get("output", ""),
            )

        return None
```

```python
"""Deep application probes plugin (Watcher agent)."""

from __future__ import annotations

import asyncio
import json
import subprocess
from datetime import datetime, timezone

import httpx
import structlog

from guardian_hc.plugin import (
    GuardianPlugin, CheckResult, HealResult, CheckContext, Severity,
)

logger = structlog.get_logger()

# Which probes run on which patrol level
PROBE_LEVELS = {
    "critical": ["jwt", "redis_deep", "celery_completion"],
    "standard": ["jwt", "redis_deep", "celery_completion", "postgres_deep", "deep_health", "pipeline", "nginx"],
    "deep": ["jwt", "redis_deep", "celery_completion", "postgres_deep", "deep_health", "pipeline", "nginx", "auth_flow"],
}


class ProbesPlugin(GuardianPlugin):
    name = "probes"
    enabled = True

    def __init__(self, config: dict):
        self._backend_url = config.get("backend_url", "http://backend:8000")
        self._redis_host = config.get("redis_host", "redis")
        self._redis_port = config.get("redis_port", 6379)
        self._redis_password = config.get("redis_password", "")
        self._pg_dsn = config.get("pg_dsn", "")
        self._nginx_url = config.get("nginx_url", "http://localhost")
        self._service_account = config.get("service_account", "guardian-probe")

    async def check(self, context: CheckContext) -> list[CheckResult]:
        probes_to_run = PROBE_LEVELS.get(context.patrol_level, PROBE_LEVELS["standard"])
        results: list[CheckResult] = []

        probe_methods = {
            "jwt": self._check_jwt,
            "redis_deep": self._check_redis_deep,
            "postgres_deep": self._check_postgres_deep,
            "deep_health": self._check_deep_health,
            "celery_completion": self._check_celery_completion,
            "pipeline": self._check_pipeline,
            "nginx": self._check_nginx,
            "auth_flow": self._check_auth_flow,
        }

        for probe_name in probes_to_run:
            method = probe_methods.get(probe_name)
            if method:
                try:
                    probe_results = await method(context)
                    results.extend(probe_results)
                except Exception as e:
                    logger.error("probe.error", probe=probe_name, error=str(e)[:200])
                    results.append(CheckResult(
                        plugin=self.name, module="Probes",
                        check_name=probe_name, status="fail",
                        severity=Severity.WARNING,
                        summary=f"Probe {probe_name} errored: {str(e)[:100]}",
                    ))

        return results

    async def heal(self, result: CheckResult) -> HealResult | None:
        hint = result.heal_hint or ""
        if hint == "restart_backend":
            from guardian_hc.healers.container_healer import ContainerHealer
            healer = ContainerHealer()
            h = await healer.heal("sowknow4-backend")
            return HealResult(
                plugin=self.name, target="sowknow4-backend",
                action="restarted", success=h.get("healed", False),
                details=str(h),
            )
        if hint == "redis_memory_purge":
            return await self._heal_redis_memory()
        if hint.startswith("restart:"):
            container = hint.split(":", 1)[1]
            from guardian_hc.healers.container_healer import ContainerHealer
            healer = ContainerHealer()
            h = await healer.heal(container)
            return HealResult(
                plugin=self.name, target=container,
                action="restarted", success=h.get("healed", False),
            )
        if hint == "kill_pg_idle":
            return await self._heal_pg_idle()
        if hint == "requeue_stuck_docs":
            return await self._heal_requeue_stuck()
        return None

    # --- Individual probes ---

    async def _check_jwt(self, ctx: CheckContext) -> list[CheckResult]:
        async with httpx.AsyncClient(timeout=10) as client:
            resp = await client.get(f"{self._backend_url}/api/v1/health/auth-check")
            data = resp.json()
            if resp.status_code == 200 and data.get("jwt_valid"):
                return [CheckResult(
                    plugin=self.name, module="Authentication Service",
                    check_name="jwt_validity", status="pass",
                    severity=Severity.INFO, summary="JWT issuance and verification OK",
                    details=data,
                )]
            return [CheckResult(

                    details={"active": active, "max": max_conn, "pct": pct},
                ))
            else:
                results.append(CheckResult(
                    plugin=self.name, module="Storage Layer",
                    check_name="pg_connections", status="pass",
                    severity=Severity.INFO,
                    summary=f"PostgreSQL connections: {active}/{max_conn} ({pct:.0f}%)",
                    details={"active": active, "max": max_conn, "pct": pct},
                ))

            # Idle-in-transaction > 5min
            idle_result = subprocess.run(
                cmd_base + [
                    "SELECT count(*) FROM pg_stat_activity "
                    "WHERE state = 'idle in transaction' "
                    "AND query_start < now() - interval '5 minutes';"

            results.append(CheckResult(
                plugin=self.name, module="Storage Layer",
                check_name="postgres_deep", status="fail",
                severity=Severity.HIGH,
```

```python
# guardian_hc/plugins/sentinel.py
"""Sentinel plugin — silent failure detection (Watcher agent)."""

from __future__ import annotations

import subprocess
from datetime import datetime, timezone, timedelta

import httpx
import structlog

from guardian_hc.plugin import (
    GuardianPlugin, CheckResult, HealResult, CheckContext, Severity,
)

logger = structlog.get_logger()

STALENESS_THRESHOLD_MINUTES = 5
QUEUE_GROWTH_CHECKS_THRESHOLD = 3  # consecutive checks showing growth


class SentinelPlugin(GuardianPlugin):
    name = "sentinel"
    enabled = True

    def __init__(self, config: dict):
        self._backend_url = config.get("backend_url", "http://backend:8000")
        self._redis_host = config.get("redis_host", "redis")
        self._redis_port = config.get("redis_port", 6379)
        self._redis_password = config.get("redis_password", "")
        # Track queue depth history for trend detection
        self._queue_history: list[int] = []

    async def check(self, context: CheckContext) -> list[CheckResult]:
        results: list[CheckResult] = []

        checks = [
            self._check_stale_data,
            self._check_queue_drain,
            self._check_frontend_api_bridge,
        ]

        for check_fn in checks:
            try:
                check_results = await check_fn(context)
                results.extend(check_results)
            except Exception as e:
                logger.error("sentinel.check.error", check=check_fn.__name__, error=str(e)[:200])

        return results

    async def heal(self, result: CheckResult) -> HealResult | None:
        hint = result.heal_hint or ""
        if hint == "restart_backend":
            from guardian_hc.healers.container_healer import ContainerHealer
            healer = ContainerHealer()
            h = await healer.heal("sowknow4-backend")
            return HealResult(
                plugin=self.name, target="sowknow4-backend",
                action="restarted", success=h.get("healed", False),
            )
        if hint == "restart_celery_workers":
            from guardian_hc.healers.container_healer import ContainerHealer
            healer = ContainerHealer()
            success = True
            for container in ["sowknow4-celery-light", "sowknow4-celery-heavy"]:
                h = await healer.heal(container)
                if not h.get("healed"):
                    success = False
            return HealResult(
                plugin=self.name, target="celery-workers",
                action="restarted", success=success,
            )
        return None

    async def _check_stale_data(self, ctx: CheckContext) -> list[CheckResult]:
        """Detect backend returning 200 but with stale data."""
        try:
            async with httpx.AsyncClient(timeout=15) as client:
                resp = await client.get(f"{self._backend_url}/api/v1/health/deep")
                if resp.status_code != 200:
                    return []  # Not our problem — probes plugin handles this

                data = resp.json()
                last_write = data.get("last_write")
                if not last_write:
                    return []  # No writes yet, can't determine staleness

                last_dt = datetime.fromisoformat(last_write)
                if last_dt.tzinfo is None:
                    last_dt = last_dt.replace(tzinfo=timezone.utc)

                age_minutes = (datetime.now(timezone.utc) - last_dt).total_seconds() / 60

                if age_minutes > STALENESS_THRESHOLD_MINUTES:
                    return [CheckResult(
                        plugin=self.name, module="Storage Layer",
                        check_name="stale_data", status="fail",
                        severity=Severity.HIGH,
                        summary=f"Last DB write was {age_minutes:.0f}min ago (threshold: {STALENESS_THRESHOLD_MINUTES}min)",
                        details={"last_write": last_write, "age_minutes": age_minutes},
                        needs_healing=True, heal_hint="restart_backend",
                    )]
                return []
        except Exception as e:
            logger.warning("sentinel.stale_data.error", error=str(e)[:200])
            return []

    async def _check_queue_drain(self, ctx: CheckContext) -> list[CheckResult]:
        """Detect queues growing but not draining."""
        try:
            cmd = ["docker", "exec", "sowknow4-redis", "redis-cli"]
            if self._redis_password:
                cmd.extend(["-a", self._redis_password])

            # Get total queue depth across known queues
            total = 0
            for queue in ["celery", "document_processing", "heavy_processing", "collection_processing"]:
                proc = subprocess.run(
                    cmd + ["LLEN", queue], capture_output=True, text=True, timeout=5,
                )
                try:
                    total += int(proc.stdout.strip().split()[-1])
                except (ValueError, IndexError):
                    pass

            self._queue_history.append(total)
            # Keep last 6 checks (1 hour at standard patrol)
            if len(self._queue_history) > 6:
                self._queue_history.pop(0)

            # Check for monotonic growth
            if len(self._queue_history) >= QUEUE_GROWTH_CHECKS_THRESHOLD:
                recent = self._queue_history[-QUEUE_GROWTH_CHECKS_THRESHOLD:]
                monotonic_growth = all(recent[i] < recent[i + 1] for i in range(len(recent) - 1))
                if monotonic_growth and recent[-1] > 10:  # Ignore trivial queues
                    return [CheckResult(
                        plugin=self.name, module="Document Pipeline",
                        check_name="queue_not_draining", status="fail",
                        severity=Severity.HIGH,
                        summary=f"Queue depth growing: {recent[0]} → {recent[-1]} over {len(recent)} checks",
                        details={"history": recent, "current": total},
                        needs_healing=True, heal_hint="restart_celery_workers",
                    )]

            return []
        except Exception as e:
            logger.warning("sentinel.queue_drain.error", error=str(e)[:200])
            return []

    async def _check_frontend_api_bridge(self, ctx: CheckContext) -> list[CheckResult]:
        """Detect frontend up but API calls returning 502/503."""
        try:
            async with httpx.AsyncClient(timeout=10) as client:
                # Check frontend is reachable
                try:
                    fe_resp = await client.get("http://frontend:3000/")
                    frontend_up = fe_resp.status_code == 200
                except Exception:
                    frontend_up = False

                if not frontend_up:
                    return []  # Frontend itself is down — not a silent failure

                # Check API through frontend proxy
                try:
                    api_resp = await client.get(f"{self._backend_url}/api/v1/health")
                    if api_resp.status_code in (502, 503):
                        return [CheckResult(
```

```python
# Metrics to track with their thresholds for prediction
DEFAULT_THRESHOLDS = {
    "disk.usage_pct": 85.0,
    "redis.memory_rss_pct": 85.0,
    "pg.connection_pct": 80.0,
    "vps.load1": 6.0,
    "vps.steal_pct": 20.0,
}

# How far ahead to predict (hours)
PREDICTION_HORIZON = 4.0

# Metrics to check for anomalies
ANOMALY_METRICS = [
    "redis.memory_rss",
    "pg.active_connections",
    "celery.total_queue_depth",
    "disk.usage_pct",
    "backend.response_ms",
]


class TrendsPlugin(GuardianPlugin):
    name = "trends"
    enabled = True

    def __init__(self, config: dict):
        self._retention_raw_hours = _parse_retention(config.get("retention_raw", "48h"))
        self._retention_hourly_days = _parse_retention(config.get("retention_hourly", "14d")) // 24
        self._redis_host = config.get("redis_host", "redis")
        self._redis_port = config.get("redis_port", 6379)
        self._redis_password = config.get("redis_password", "")
        self._thresholds = dict(DEFAULT_THRESHOLDS)

    async def check(self, context: CheckContext) -> list[CheckResult]:
        """Collect metrics and write to DB. Returns empty results (collection only)."""
        db: MetricsDB | None = context.metrics_db
        if not db:
            return []

        all_metrics: list[tuple[str, float, str, dict]] = []

        collectors = [
            self._collect_host_metrics,
            self._collect_redis_metrics,
            self._collect_pg_metrics,
            self._collect_celery_metrics,
            self._collect_backend_metrics,
        ]

        for collector in collectors:
            try:
                metrics = await collector()
                all_metrics.extend(metrics)
            except Exception as e:
                logger.warning("trends.collect.error", collector=collector.__name__, error=str(e)[:200])

        if all_metrics:
            try:
                await db.write_batch(all_metrics)
                logger.debug("trends.collected", count=len(all_metrics))
            except Exception as e:
                logger.error("trends.write.error", error=str(e)[:200])

        return []

    async def analyze(self, context: AnalysisContext) -> list[Insight]:
        """Run trend analysis: slope detection, anomaly detection, capacity alerts."""
        db: MetricsDB | None = context.metrics_db
        if not db:
            return []

        insights: list[Insight] = []

        # Slope-based prediction
        for metric, threshold in self._thresholds.items():
            try:
                slope = await db.get_slope(metric, hours=6)
                if slope is None or slope <= 0:
                    continue

                current = await db.get_latest(metric)
                if current is None:
                    continue

                if current >= threshold:
                    continue  # Already breached, not a prediction

                hours_to_breach = (threshold - current) / slope
                if 0 < hours_to_breach <= PREDICTION_HORIZON:
                    insights.append(Insight(
                        plugin=self.name, insight_type="prediction",
                        severity=Severity.WARNING,
                        summary=f"{metric} will hit {threshold} in ~{hours_to_breach:.1f}h at current rate",
                        metric=metric,
                        current_value=current,
                        predicted_value=threshold,
                        predicted_time_hours=hours_to_breach,
                        recommended_action=_action_for_metric(metric),
                    ))
            except Exception as e:
                logger.warning("trends.slope.error", metric=metric, error=str(e)[:200])

        # Anomaly detection
        for metric in ANOMALY_METRICS:
            try:
                current = await db.get_latest(metric)
                if current is None:
                    continue

                mean, std = await db.get_mean_stddev(metric, hours=24)
                if std == 0:
                    continue

                z_score = abs(current - mean) / std
                if z_score > 2.0:
                    insights.append(Insight(
                        plugin=self.name, insight_type="anomaly",
                        severity=Severity.WARNING,
                        summary=f"{metric} is anomalous: {current:.1f} (mean: {mean:.1f}, z: {z_score:.1f})",
                        metric=metric,
                        current_value=current,
                    ))
            except Exception as e:
                logger.warning("trends.anomaly.error", metric=metric, error=str(e)[:200])

        return insights

    """Parse retention string like '48h' or '14d' into hours."""
    s = s.strip().lower()
    if s.endswith("h"):
        return int(s[:-1])
    if s.endswith("d"):
        return int(s[:-1]) * 24
    return int(s)

        "pg.connection_pct": "kill_idle_transactions",
        "vps.load1": "alert_only",
        "vps.steal_pct": "alert_only",
    }
    return actions.get(metric, "alert_only")
```

```python
"""Memory plugin — incident learning engine (Learner agent)."""

from __future__ import annotations

import json
from pathlib import Path

import structlog

from guardian_hc.plugin import (
    GuardianPlugin, CheckResult, HealResult, Insight,
    CheckContext, AnalysisContext, Severity,
)
from guardian_hc.db import MetricsDB

logger = structlog.get_logger()

CONFIDENCE_THRESHOLD = 0.7  # Only act on high-confidence patterns


class PatternMatcher:
    """Evaluates current metrics against learned patterns."""

    def __init__(self, db: MetricsDB):
        self._db = db

    async def evaluate(self) -> list[dict]:
        """Return patterns whose trigger conditions match current metrics."""
        patterns = await self._db.get_active_patterns()
        matches = []

        for pattern in patterns:
            if pattern["confidence"] < CONFIDENCE_THRESHOLD:
                continue

            conditions = pattern["trigger_conditions"]
            if isinstance(conditions, str):
                conditions = json.loads(conditions)

            all_match = True
            for metric, rule in conditions.items():
                current = await self._db.get_latest(metric)
                if current is None:
                    all_match = False
                    break

                if isinstance(rule, dict):
                    if ">" in rule and current <= rule[">"]:
                        all_match = False
                        break
                    if "<" in rule and current >= rule["<"]:
                        all_match = False
                        break
                    if "==" in rule and current != rule["=="]:
                        all_match = False
                        break

            if all_match:
                matches.append(pattern)

        return matches


class MemoryPlugin(GuardianPlugin):
    name = "memory"
    enabled = True

    def __init__(self, config: dict):
        self._bootstrap_sources = config.get("bootstrap_sources", [])
        self._bootstrapped = False

        return []  # Memory plugin doesn't run checks

        db: MetricsDB | None = context.metrics_db or context.patterns_db
        if not db:
            return []

        # Bootstrap patterns on first deep patrol
        if not self._bootstrapped:
            await self._bootstrap(db)
            self._bootstrapped = True

        # Evaluate patterns against current metrics
        matcher = PatternMatcher(db)
        matches = await matcher.evaluate()

        insights = []
        for pattern in matches:

                pattern_name=pattern["pattern_name"],
                confidence=pattern["confidence"],
                recommended_action=pattern["recommended_action"],
            ))
            logger.info(
                "memory.pattern_matched",
                pattern=pattern["pattern_name"],
                confidence=pattern["confidence"],
                action=pattern["recommended_action"],
            )

        return insights

    async def _bootstrap(self, db: MetricsDB):

        existing = await db.get_active_patterns()
```

```python
class TelegramCommandHandler:
    """Handles /guardian commands via Telegram Bot API polling."""

    def __init__(self, guardian, token: str, chat_id: str):
        self._guardian = guardian
        self._token = token
        self._chat_id = chat_id
        self._last_update_id = 0
        self._commands = {
            "/guardian": self._cmd_status,
            "/gstatus": self._cmd_status,
            "/gtrends": self._cmd_trends,
            "/gprobes": self._cmd_probes,
            "/gpatterns": self._cmd_patterns,
            "/gincidents": self._cmd_incidents,
            "/glearn": self._cmd_learn,
            "/gheal": self._cmd_heal,
            "/gsilence": self._cmd_silence,
        }

    async def poll_once(self):
        """Check for new commands and respond."""
        try:
            url = f"https://api.telegram.org/bot{self._token}/getUpdates"
            async with httpx.AsyncClient(timeout=10) as client:
                resp = await client.get(url, params={
                    "offset": self._last_update_id + 1,
                    "timeout": 5,
                    "allowed_updates": ["message"],
                })
                data = resp.json()
                for update in data.get("result", []):
                    self._last_update_id = update["update_id"]
                    msg = update.get("message", {})
                    text = msg.get("text", "")
                    chat_id = str(msg.get("chat", {}).get("id", ""))

                    if chat_id != self._chat_id:
                        continue

                    parts = text.strip().split()
                    cmd = parts[0].lower() if parts else ""
                    args = parts[1:] if len(parts) > 1 else []

                    handler = self._commands.get(cmd)
                    if handler:
                        response = await handler(args)
                        await self._send(response)
        except Exception as e:
            logger.debug("telegram.poll.error", error=str(e)[:200])

    async def _send(self, text: str):
        url = f"https://api.telegram.org/bot{self._token}/sendMessage"
        async with httpx.AsyncClient(timeout=10) as client:
            await client.post(url, json={
                "chat_id": self._chat_id,
                "text": text[:4096],
                "parse_mode": "HTML",
            })
```

```python
def svg_sparkline(values: list[float], width: int = 200, height: int = 40, color: str = "#3b82f6") -> str:
    """Generate an inline SVG sparkline from a list of values."""
    if not values or len(values) < 2:
        return ""
    min_v = min(values)
    max_v = max(values)
    range_v = max_v - min_v or 1
    step = width / (len(values) - 1)
    points = []
    for i, v in enumerate(values):
        x = i * step
        y = height - ((v - min_v) / range_v) * (height - 4) - 2
        points.append(f"{x:.1f},{y:.1f}")
    polyline = " ".join(points)
    return (
        f'<svg width="{width}" height="{height}" xmlns="http://www.w3.org/2000/svg">'
        f'<polyline points="{polyline}" fill="none" stroke="{color}" stroke-width="1.5"/>'
        f'</svg>'
    )
```
